\[\begin{array}{l}
(Ax,x)=(A_1x,x)=\mathlarger{\int}_{\mathbb{R}}zd||E(z)x||^2

\[\begin{array}{rl}
(A_1x,x)&= K\mathlarger{\int}_{\mathbb{R}}zd||E(z)x||^2=K\mathlarger{\int}_{(-\infty,K]}d||E(z)x||^2+K\mathlarger{\int}_{(K,+\infty]}d||E(z)x||^2\\
